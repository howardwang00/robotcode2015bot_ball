#define RED 0
#define GREEN 1
int create_track_mesa()
{
	int area = 0;
	float time = curr_time();
	while(time+3>curr_time())
	{
		camera_update();
		area = get_object_area(RED,0);
		if(area>500)
		{
			printf("Seen Blob of Red color\n");
			return RED;
		}
		area = get_object_area(GREEN,0);
		if(area>500)
		{
			printf("Seen Blob of Green color\n");
			return GREEN
		}
	}
}
